A simple Wordle-style word guessing game. Play below.

<div style="min-width:220px;" markdown="1">
**How to play**
- Guess the hidden word.
- <span style="display:inline-block;padding:0 6px;margin:0 2px;background:#e0e0e0;border-radius:4px;font-weight:bold;">←</span> / <span style="display:inline-block;padding:0 6px;margin:0 2px;background:#e0e0e0;border-radius:4px;font-weight:bold;">→</span> to decrease or increase word length (3–7 letters).
- Type a letter to fill the current row; **Backspace** to remove the last letter; **Enter** to submit a guess.
- After each guess, the colors show how close you were:
  - <span style="color:#6aaa64;font-weight:bold;"><span style="display:inline-block;width:14px;height:14px;background:#6aaa64;margin-right:6px;vertical-align:middle;border-radius:2px;"></span>Green</span> — letter is in the word and in the correct spot.
  - <span style="color:#b59f3b;font-weight:bold;"><span style="display:inline-block;width:14px;height:14px;background:#c9b458;margin-right:6px;vertical-align:middle;border-radius:2px;"></span>Yellow</span> — letter is in the word but in the wrong spot.
  - <span style="color:#787c7e;font-weight:bold;"><span style="display:inline-block;width:14px;height:14px;background:#787c7e;margin-right:6px;vertical-align:middle;border-radius:2px;"></span>Gray</span> — letter is not in the word.

class WordGuessingGame {
  constructor(options = {}) {
    const opts = this._parseOptions(options);
    this.numberOfTries = opts.numberOfTries;
    this.numberOfLetters = opts.numberOfLetters;
    this.containerId = opts.containerId;
  }

  async start() {
    if (this._boundKeyHandler) {
      document.removeEventListener('keydown', this._boundKeyHandler);
    }
    this._ended = false;
    this.targetWord = await this._pickTargetWord(this.numberOfLetters);
    this.currentRowIndex = 0;
    this.currentColumnIndex = 0;

    const container = this._getContainer(this.containerId);
    const { grid, rows } = this._buildGrid(this.numberOfTries, this.numberOfLetters);
    this.grid = grid;
    this.rows = rows;
    container.appendChild(grid);

    const msgEl = document.createElement('div');
    msgEl.id = this.containerId + '-msg';
    msgEl.style.marginTop = '8px';
    msgEl.style.minHeight = '1.4em';
    msgEl.style.fontSize = '0.9rem';
    msgEl.style.color = '#c62828';
    container.appendChild(msgEl);

    const showAnswerWrap = document.createElement('div');
    showAnswerWrap.style.marginTop = '10px';
    showAnswerWrap.style.fontSize = '0.85rem';
    const showBtn = document.createElement('button');
    showBtn.textContent = 'Show answer';
    showBtn.style.padding = '4px 10px';
    showBtn.style.fontSize = '0.8rem';
    showBtn.style.cursor = 'pointer';
    showBtn.style.borderRadius = '4px';
    showBtn.style.border = '1px solid #d3d6da';
    showBtn.style.background = '#f5f5f5';
    const answerSpan = document.createElement('span');
    answerSpan.style.marginLeft = '8px';
    answerSpan.style.fontWeight = 'bold';
    showBtn.addEventListener('click', () => {
      answerSpan.textContent = this.targetWord.charAt(0).toUpperCase() + this.targetWord.slice(1);
    });
    showAnswerWrap.appendChild(showBtn);
    showAnswerWrap.appendChild(answerSpan);
    container.appendChild(showAnswerWrap);

    this._attachListeners();
  }

  _attachListeners() {
    this._boundKeyHandler = this._onKeyDown.bind(this);
    document.addEventListener('keydown', this._boundKeyHandler);
  }

  _onKeyDown(e) {
    // allow refresh/reload page
    if ((e.ctrlKey || e.metaKey) && e.key === 'r') return;
    if (e.key === 'F5') return;
    if (e.key === 'ArrowRight') {
      e.preventDefault();
      if (this.numberOfLetters < 7) {
        this.numberOfLetters++;
        this.start();
      }
      return;
    }
    if (e.key === 'ArrowLeft') {
      e.preventDefault();
      if (this.numberOfLetters > 3) {
        this.numberOfLetters--;
        this.start();
      }
      return;
    }
    if (this._ended) {
      if (e.key === 'Enter') {
        e.preventDefault();
        this.start();
      }
      return;
    }
    const row = this.rows[this.currentRowIndex];
    if (!row) return;

    if (e.key === 'Backspace') {
      e.preventDefault();
      if (this.currentColumnIndex > 0) {
        this.currentColumnIndex--;
        row[this.currentColumnIndex].textContent = '';
      }
      return;
    }

    if (e.key === 'Enter') {
      e.preventDefault();
      this._submitRow();
      return;
    }

    this._clearMessage();

    if (e.key.length === 1 && /^[a-zA-Z]$/.test(e.key)) {
      e.preventDefault();
      if (this.currentColumnIndex < this.numberOfLetters) {
        row[this.currentColumnIndex].textContent = this.currentColumnIndex === 0 ? e.key.toUpperCase() : e.key.toLowerCase();
        this.currentColumnIndex++;
      }
    }
  }

  async _submitRow() {
    if (this.currentColumnIndex !== this.numberOfLetters) {
      this._flashEmptyCells();
      return;
    }
    const row = this.rows[this.currentRowIndex];
    const guess = row.map(cell => cell.textContent).join('').toLowerCase();
    const valid = await this._isValidWord(guess);
    if (!valid) {
      this._showMessage('Not a valid word');
      return;
    }
    this._clearMessage();
    const feedback = this._computeFeedback(guess);
    this._applyRowFeedback(row, feedback);

    if (guess === this.targetWord) {
      this._endGame(true);
      return;
    }
    if (this.numberOfTries === this.currentRowIndex + 2) {
      this.numberOfTries++;
      const newRowIndex = this.rows.length;
      const cells = [];
      for (let c = 0; c < this.numberOfLetters; c++) {
        const cell = this._createCell(newRowIndex, c);
        cells.push(cell);
        this.grid.appendChild(cell);
      }
      this.rows.push(cells);
      cells[0].scrollIntoView({ behavior: 'smooth', block: 'nearest' });
    }

    this.currentRowIndex++;
    this.currentColumnIndex = 0;
  }

  _messageEl() {
    return document.getElementById(this.containerId + '-msg');
  }

  _showMessage(text) {
    this._messageEl().textContent = text;
  }

  _clearMessage() {
    const el = document.getElementById(this.containerId + '-msg');
    if (el) el.textContent = '';
  }

  _flashEmptyCells() {
    const row = this.rows[this.currentRowIndex];
    for (let c = this.currentColumnIndex; c < this.numberOfLetters; c++) {
      const cell = row[c];
      cell.style.backgroundColor = '#f8b4b4';
      setTimeout(() => {
        cell.style.backgroundColor = '';
      }, 200);
    }
  }

  _computeFeedback(guess) {
    const target = this.targetWord;
    const n = target.length;
    const result = Array(n).fill(null);

    for (let i = 0; i < n; i++) {
      if (guess[i] === target[i]) {
        result[i] = 'green';
      } else if (target.includes(guess[i])) {
        result[i] = 'yellow';
      } else {
        result[i] = 'red';
      }
    }
    return result;
  }

  _applyRowFeedback(row, feedback) {
    const colors = { green: '#6aaa64', yellow: '#c9b458', red: '#787c7e' };
    row.forEach((cell, i) => {
      cell.style.backgroundColor = colors[feedback[i]];
      cell.style.borderColor = colors[feedback[i]];
      cell.style.color = '#fff';
    });
  }

  _endGame(won) {
    this._ended = true;
    if (!won) document.removeEventListener('keydown', this._boundKeyHandler);
    const container = document.getElementById(this.containerId);
    const msg = document.createElement('div');
    msg.style.marginTop = '12px';
    msg.style.fontWeight = 'bold';
    msg.style.fontSize = '1.25rem';
    msg.textContent = won ? 'You win!' : `You lose. The word was: ${this.targetWord}`;
    container.appendChild(msg);
    if (won) {
      const again = document.createElement('div');
      again.style.marginTop = '8px';
      again.style.fontWeight = 'normal';
      again.style.fontSize = '0.95rem';
      again.textContent = 'Press Enter to play again.';
      container.appendChild(again);
    }
  }

  _parseOptions(options) {
    const containerId = options.containerId || 'wordle';
    if (!containerId) throw new Error('WordGuessingGame: containerId is required');
    const numberOfLetters = options.numberOfLetters || 4;
    const list = WORD_LISTS[numberOfLetters];
    if (!list || !list.length) throw new Error('WordGuessingGame: no word list for length ' + numberOfLetters);
    return {
      numberOfTries: options.numberOfTries || 6,
      numberOfLetters,
      containerId
    };
  }

  async _pickTargetWord(numberOfLetters) {
    try {
      const res = await fetch(`https://random-word-api.herokuapp.com/word?length=${numberOfLetters}`);
      if (res.ok) {
        const words = await res.json();
        if (Array.isArray(words) && words.length > 0 && String(words[0]).length === numberOfLetters) {
          return String(words[0]).toLowerCase();
        }
      }
    } catch (_) { /* fallback */ }
    const list = WORD_LISTS[numberOfLetters];
    return list[Math.floor(Math.random() * list.length)].toLowerCase();
  }

  async _isValidWord(word) {
    try {
      const res = await fetch(`https://api.dictionaryapi.dev/api/v2/entries/en/${encodeURIComponent(word)}`);
      return res.ok;
    } catch (_) {
      return false;
    }
  }

  _getContainer(containerId) {
    const container = document.getElementById(containerId);
    if (!container) throw new Error('WordGuessingGame: container not found: ' + containerId);
    container.innerHTML = '';
    return container;
  }

  _createCell(row, col) {
    const cell = document.createElement('div');
    cell.setAttribute('data-row', row);
    cell.setAttribute('data-col', col);
    cell.style.border = '1px solid #d3d6da';
    cell.style.backgroundColor = '#f5f5f5';
    cell.style.aspectRatio = '1';
    cell.style.display = 'flex';
    cell.style.alignItems = 'center';
    cell.style.justifyContent = 'center';
    cell.style.fontSize = '1.5rem';
    cell.style.fontWeight = 'bold';
    return cell;
  }

  _buildGrid(numberOfTries, numberOfLetters) {
    const grid = document.createElement('div');
    grid.style.display = 'grid';
    grid.style.gridTemplateColumns = `repeat(${numberOfLetters}, 1fr)`;
    grid.style.gap = '4px';
    grid.style.maxWidth = `${numberOfLetters * 52}px`;

    const rows = [];
    for (let r = 0; r < numberOfTries; r++) {
      const cells = [];
      for (let c = 0; c < numberOfLetters; c++) {
        const cell = this._createCell(r, c);
        cells.push(cell);
        grid.appendChild(cell);
      }
      rows.push(cells);
    }
    return { grid, rows };
  }
}

document.addEventListener('DOMContentLoaded', () => {
  const game = new WordGuessingGame({ containerId: 'wordle' });
  game.start();
});
</script>
