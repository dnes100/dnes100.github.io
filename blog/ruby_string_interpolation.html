~ Ruby string #{interpolation} with arrays! 

          def generate_intro_letter(options = {})
            recipient = options[:recipient]
            messages = options[:messages]
            name = options[:name]

            [
              'Hello',
              (recipient),
              ('!,' if recipient),
              '\n\n',
              (messages.join('\n') if messages),
              ('\n\n' if messages),
              'Thanks :)',
              '\n',
              (name if name),
            ].compact.join(' ')
          end
